import json
from js import document, window, console
from pyodide.http import pyfetch
from pyodide.ffi import create_proxy

async def refine_text(event):
    api_key = document.getElementById("apiKey").value.strip()
    input_text = document.getElementById("textInput").value.strip()
    
    run_btn = document.getElementById("runBtn")
    status = document.getElementById("aiStatus")
    result_card = document.getElementById("resultCard")
    error_card = document.getElementById("errorCard")
    error_msg_area = document.getElementById("errorMsg")
    output_area = document.getElementById("outputArea")

    result_card.style.display = "none"
    error_card.style.display = "none"
    
    if not api_key:
        error_msg_area.textContent = "API Key가 입력되지 않았습니다."
        error_card.style.display = "block"
        return
    if not input_text:
        error_msg_area.textContent = "내용을 입력해주세요."
        error_card.style.display = "block"
        return

    run_btn.disabled = True
    status.style.display = "block"

    # [수정] 모델명을 models : gemini-2.5-flash 로 변경
    url = f"https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key={api_key}"
    
    prompt = (
        "Role: Professional Business Editor\n"
        "Task: Refine the text into professional business language.\n"
        "Constraint 1: Keep the EXACT same structure, delimiters (like -, :, ;, |), and spacing.\n"
        "Constraint 2: Do NOT change the layout or position of lines/words.\n"
        "Constraint 3: Only replace words/phrases with more formal business terms.\n"
        "Constraint 4: Output ONLY the refined text. No markdown, no notes.\n\n"
        f"Text to refine:\n{input_text}"
    )

    payload = {"contents": [{"parts": [{"text": prompt}]}]}

    try:
        response = await pyfetch(
            url=url,
            method="POST",
            headers={"Content-Type": "application/json"},
            body=json.dumps(payload)
        )
        
        if response.status == 200:
            data = await response.json()
            try:
                refined_text = data["candidates"][0]["content"]["parts"][0]["text"]
                output_area.textContent = refined_text.strip()
                result_card.style.display = "block"
                result_card.scrollIntoView({"behavior": "smooth"})
            except (KeyError, IndexError):
                error_msg_area.textContent = "AI가 응답을 생성했으나 해석할 수 없습니다. (Safety Filter 차단 가능성)"
                error_card.style.display = "block"
        else:
            err_json = await response.json()
            msg = err_json.get("error", {}).get("message", "Unknown Error")
            st = err_json.get("error", {}).get("status", "Unknown Status")
            
            error_msg_area.textContent = (
                f"API 오류 발생: {st}\n"
                f"메시지: {msg}\n\n"
                "체크리스트:\n"
                "1. API 키가 유효한지 확인하세요.\n"
                "2. 인터넷 연결 상태를 확인하세요.\n"
                "3. 여전히 NOT_FOUND가 발생한다면, Google AI Studio에서 새 키를 발급받는 것을 추천합니다."
            )
            error_card.style.display = "block"
            error_card.scrollIntoView({"behavior": "smooth"})
            
    except Exception as e:
        error_msg_area.textContent = f"연결 문제 발생: {str(e)}"
        error_card.style.display = "block"
    
    finally:
        run_btn.disabled = False
        status.style.display = "none"

run_click_proxy = create_proxy(refine_text)
document.getElementById("runBtn").addEventListener("click", run_click_proxy)

loading = document.getElementById("py-loading")
loading.classList.add("hidden")
window.setTimeout(create_proxy(lambda: loading.remove()), 450)
